# Comandos Utilizados nos Exemplos
docker compose up --build -d
docker compose exec app bash
npm init -y					 	  : Cria o package.json
npm install typescript --save-dev : 
npx tsc --init					  : 
npm install nodemon --save-dev    : Recarrega o server quando necessario
npm audit fix --force			  : Atualiza as dependencias para versoes mais recentes
npm install express				  : Servidor web para node.js
npm i --save-dev @types/express   : Instala a tipagem do typescript
npm instal ts-node --save-dev	  : Instala a dependencia do tsnode
npm start						  : Sobe o projeto
npm audit --production
nvm install 20
nvm use 20.20.0
